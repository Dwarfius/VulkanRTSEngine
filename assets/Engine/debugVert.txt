#version 420

layout(location = 0) in vec3 position;
layout(location = 1) in vec3 color;

layout (std140, binding = 0) uniform CameraAdapter
{
	mat4 CamViewProj;
	mat4 CamView;
	mat4 CamProj;
	vec4 FrustumPlane[6]; // left, right, down, up, far, near
	vec3 CamPos;
	vec2 Viewport;
};

out vec3 colorOut;

void main()
{
	gl_Position = CamViewProj * vec4(position, 1.0);
	colorOut = color;
}
